# cat test.c
#include <linux/module.h>
#include <linux/preempt.h>

int test_init(void)
{
  int i;

  i = preempt_count();
  barrier();
  printk ("%d\n", i);
  return i;
}
module_init(test_init);

void test_exit(void)
{
}
module_exit(test_exit);

MODULE_LICENSE("GPL");
# cat Makefile
obj-m += test.o

KVER := $(shell uname -r)
KDIR := /lib/modules/$(KVER)/build/
PWD := $(shell pwd)

all:
    make -C $(KDIR) M=$(PWD) modules

clean:
    make -C $(KDIR) M=$(PWD) clean
    rm -f *~ *.E
